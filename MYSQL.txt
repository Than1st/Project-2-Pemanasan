Join
SELECT mahasiswa.nama,
       matkul.nama_matkul,
       nilai.nilai
FROM nilai
 INNER JOIN mahasiswa ON mahasiswa.nim = nilai.fk_nim
 INNER JOIN matkul ON matkul.kd_matkul = nilai.fk_kd_mk
 ORDER BY nilai.nilai
